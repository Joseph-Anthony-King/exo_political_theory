building_exo_political_theory = {

	base_buildtime = @b2_time

	category = government

	base_cap_amount = 1

	can_build = yes
	can_demolish = yes
	can_be_ruined = yes
	can_be_disabled = yes
	position_priority = 1

	building_sets = {
		government
	}

	potential = {
		exists = owner
	}
	
	prerequisites = {
		"tech_exo_political_theory_1"
	}

	allow = {
		pop_amount >= 2000
		is_capital = yes
	}

	empire_limit = {
		base = 1
	}

	destroy_trigger = {
		OR = {
			AND = {
				exists = owner
				owner = { is_regular_empire = no }
			}
			planet = { is_capital = no }
		}
	}

    triggered_planet_modifier = {
        potential = {
            exists = owner
            owner = { is_regular_empire = yes }
        }
        modifier = {
            job_unifier_add = 100
			job_bureaucrat_add = 200
        }
    }

	country_modifier = {
		envoys_add = 1
	}
		
	resources = {
		category = planet_buildings
		produces = {
			trigger = {
				exists = owner
				owner = { 
					NOT = { 
						OR = {
							has_deficit = energy 
							has_deficit = consumer_goods 
						}
					} 
				}
			}
			influence = 2
		}		
		upkeep = {
			energy = 12
			consumer_goods = 12
		}
		cost = {
			minerals = 5000
		}
	}
}
